# Your solution HERE !!!!
def repeated_element(array):
    vistos = set()
    for numero in array:
        if numero in vistos:
            return numero
        vistos.add(numero)

# Your solution HERE !!!!
def first_last_uppercase(words):
    serie = pd.Series(words)
    return serie.apply(lambda s: s[0].upper() + s[1:-1] + s[-1].upper() if len(s) > 1 else s.upper())

# Your solution HERE !!!!
def fibonacci_generator():
    a, b = 0, 1
    while True:
        yield a
        a, b = b, a + b
